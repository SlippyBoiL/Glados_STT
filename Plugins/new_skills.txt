# DESCRIPTION: Detailed system diagnostics and hardware specs.
# --- GLADOS SKILL: skill_specs.py ---

import platform
import os
import sys

def system_diagnostic():
    try:
        uname = platform.uname()
        info = (
            f"--- SYSTEM DIAGNOSTIC ---\n"
            f"OS: {uname.system} {uname.release} (v{uname.version})\n"
            f"Node Name: {uname.node}\n"
            f"Machine: {uname.machine}\n"
            f"Processor: {uname.processor}\n"
            f"Python Version: {sys.version.split()[0]}\n"
            f"Logical Cores: {os.cpu_count()}\n"
            f"-------------------------"
        )
        return info
    except Exception as e:
        return f"Diagnostic failure: {e}"

if __name__ == "__main__":
    print(system_diagnostic())


# DESCRIPTION: Generates secure, high-entropy passwords.
# --- GLADOS SKILL: skill_passgen.py ---

import secrets
import string

def generate_password(length=16):
    try:
        alphabet = string.ascii_letters + string.digits + "!@#$%^&*"
        password = ''.join(secrets.choice(alphabet) for i in range(length))
        return f"Generated Secure String: {password}"
    except:
        return "Entropy failure."

if __name__ == "__main__":
    print(generate_password())

import subprocess

def list_processes():
    try:
        # Uses Windows tasklist
        cmd = 'tasklist /NH /FO CSV'
        output = subprocess.check_output(cmd, shell=True).decode('utf-8')
        lines = output.split('\r\n')
        # Just grab the first 10 strictly for display
        preview = "\n".join(lines[:10])
        return f"Active Processes (Snapshot):\n{preview}"
    except:
        return "Process tracking failed."

if __name__ == "__main__":
    print(list_processes())

# DESCRIPTION: Checks for open ports on localhost (Common ports).
# --- GLADOS SKILL: skill_port_check.py ---

import socket

def check_ports():
    target = "127.0.0.1"
    ports = [21, 22, 80, 443, 3306, 8080]
    results = []
    
    for port in ports:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(0.5)
        result = sock.connect_ex((target, port))
        state = "OPEN" if result == 0 else "CLOSED"
        results.append(f"Port {port}: {state}")
        sock.close()
        
    return "\n".join(results)

if __name__ == "__main__":
    print(check_ports())
